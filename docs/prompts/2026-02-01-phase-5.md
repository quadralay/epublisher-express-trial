```
$skin_background_color: #ffffff;    // Toolbar background color
$secondary_brand_color: #eeeeee;    // Secondary brand color
$secondary_text_color: #fefefe;     // Secondary text color
$tertiary_brand_color: #231f20;     // Tertiary (accent) brand color

```
